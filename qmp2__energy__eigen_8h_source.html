99 double compute (){
100 //Setting um the Vector and Matrix Maps to use Eigen methods
101 using MatMap = Map<Matrix <double, Dynamic, Dynamic, RowMajor>>;
102 using VecMap = Map<VectorXd>;
103
104 // Map memory pointer to Eigen Maps
105 MatMap map_mmo(mmo, m3Dnpts, mnocc); // occupied orbitals
106 MatMap map_mmv(mmv, m3Dnpts, mnvirt); // virtual orbitals
107 MatMap map_mm1Deps_o(mm1Deps_o, m1Dnpts, mnocc); // prefactors
108 MatMap map_mm1Deps_v(mm1Deps_v, m1Dnpts, mnvirt); // prefactors
109
110 // Create vector of maps of three dimensional 1Deps_ov
111 vector<MatMap> map_mm1Deps_ov; // prefactors
112 map_mm1Deps_ov.reserve(m1Dnpts);
113 for (size_t i = 0; i < m1Dnpts; i++){
114 map_mm1Deps_ov.push_back(
115 MatMap(mm1Deps_ov + i * mnocc * mnvirt, mnocc, mnvirt));
116 }
117
118 VecMap map_mvf(mvf, mnocc+mnvirt); // orbital energies
119 VecMap map_mv1Dpts(mv1Dpts, m1Dnpts); // 1D grid points
120 VecMap map_mv1Dwts(mv1Dwts, m1Dnpts); // 1D grid point weights
121
122 VectorXd o_p; //empty Vectors, so that each OpenMP thread
123 VectorXd v_p; // can use a private copy
124 MatrixXd c2_p;
125
126 double energy = 0;
127#pragma omp parallel for reduction(+:energy) schedule(dynamic) default(none)\
128 private(o_p, v_p, c2_p)\
129 shared(moffset, mnpts_to_proc, m1Dnpts, m3Dnpts, mnocc, mnvirt, \
130 map_mv1Dwts, map_mmo, map_mm1Deps_o, map_mmv, \
131 map_mm1Deps_v, mc_c, map_mm1Deps_ov)\
132 collapse(2)
133 // loop over points on 1D grid
134 for (int k = 0; k < m1Dnpts; k++){
135 // loop over all points on 3D grid in set of elements
136 // to compute from offset position
137 for (int p = moffset; p < moffset + mnpts_to_proc; p++){
138 // create local copy of with prefactor scaled orbitals
139 o_p = map_mmo.row(p)
140 .cwiseProduct(map_mm1Deps_o.row(k));
141 v_p = map_mmv.row(p)
142 .cwiseProduct(map_mm1Deps_v.row(k));
143 c2_p = MatMap(mc_c + p * mnocc * mnvirt, mnocc, mnvirt)
144 .cwiseProduct(map_mm1Deps_ov[k]);
145
146 // loop over inner grid points, utilizing symmetry
147 for (int q = 0; q <= p; q++){
148 // local copy of orbitals to minimize transpose
149 VectorXd o_q = map_mmo.row(q).transpose();
150 VectorXd v_q = map_mmv.row(q).transpose();
151 MatMap map_mc_c_q(mc_c + q * mnocc * mnvirt,
152 mnocc, mnvirt);
153
154 double jo = 0;
155 for (int a = 0; a < mnvirt; a++){
156 double tmp1 = o_p.dot(map_mc_c_q.col(a));
157 double tmp2 = o_q.dot(c2_p.col(a));
158 jo += tmp1 * tmp2;
159 }//a
160
161 double j = (c2_p.cwiseProduct(map_mc_c_q)).sum();
162
163 double o = (o_p).dot(o_q);
164
165 double v = (v_p).dot(v_q);
166
167 double sum = (jo - 2 * j * o) * v;
168
169 // multiplication of p,q pair function is commutative
170 // so we calculate only unique pairs, so only half
171 // of the duples and double them
172 if (p != q){
173 sum *= 2.0;
174 }//if
175
176 // weight the energy contribution of 1D grid
177 // point
178 energy += map_mv1Dwts(k) * sum;
179 }//for q
180 }//p
181 }//k
182
183 return energy;
184 };
