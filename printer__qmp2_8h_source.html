161 string make_line(string in, char align){
162 size_t length_in = in.length();
163 string out = "";
164
165 // Calculating overal space
166 size_t space = ((length_in < (mwidth - 2)) ?
167 ((mwidth - 2) - length_in) : 0);
168 string left = "";
169 string right = "";
170
171 if (align == 'l'){
172 left = ((space > 0 ) ? " " : "" );
173 space--;
174 for (size_t i = 0; i < space; i++){
175 right += " ";
176 }
177 } else if (align == 'r'){
178 left = ((space > 0 ) ? " " : "" );
179 space--;
180 for (size_t i = 0; i < space; i++){
181 left += " ";
182 }
183
184 } else {
185 for (size_t i = space; i > 1; i -= 2){
186 left += " ";
187 right += " ";
188 }
189 left += ((space % 2) == 1) ? " " : "";
190 }
191
192 //finish out
193 out = "|" + left + in + right + "|";
194 return out;
195 };
