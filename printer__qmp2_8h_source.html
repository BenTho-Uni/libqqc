168 string make_line(string in, char align){
169 size_t length_in = in.length();
170 string out = "";
171
172 // Calculating overal space
173 size_t space = ((length_in < (mwidth - 2)) ?
174 ((mwidth - 2) - length_in) : 0);
175 string left = "";
176 string right = "";
177
178 if (align == 'l'){
179 left = ((space > 0 ) ? " " : "" );
180 space--;
181 for (size_t i = 0; i < space; i++){
182 right += " ";
183 }
184 } else if (align == 'r'){
185 left = ((space > 0 ) ? " " : "" );
186 space--;
187 for (size_t i = 0; i < space; i++){
188 left += " ";
189 }
190
191 } else {
192 for (size_t i = space; i > 1; i -= 2){
193 left += " ";
194 right += " ";
195 }
196 left += ((space % 2) == 1) ? " " : "";
197 }
198
199 //finish out
200 out = "|" + left + in + right + "|";
201 return out;
202 };
