56 bool check_data_validity() {
57
58 if (mnocc == 0) throw invalid_argument(
59 "Number of occupied orbitals shouldn't be 0.");
60 if (mnmo == 0) throw invalid_argument(
61 "Number of molecular orbitals shouldn't be 0.");
62 if (mnao == 0) throw invalid_argument(
63 "Number of atomic orbitals shouldn't be 0.");
64 if (!m1Dgrid.check_data_validity()) throw invalid_argument(
65 "1D Grid data not valid.");
66 if (!m3Dgrid.check_data_validity()) throw invalid_argument(
67 "3D Grid data not valid.");
68 if (!mmat_fock) throw invalid_argument(
69 "Fock matrix pointer cannot be NULL.");
70 if (!mmat_cgto) throw invalid_argument(
71 "Cgto matrix pointer cannot be NULL.");
72 if (!mcube_coul) throw invalid_argument(
73 "Coulomb integral matrix pointer cannot be NULL.");
74
75 return true;
