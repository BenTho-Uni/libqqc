99 int prnt_lvl, Grid p1Dgrid, Grid p3Dgrid, double* mat_fock,
100 double* mat_coeff, double* mat_cgto, double* cube_coul) :
101 mnocc(nocc), mnvirt(nvirt), mnao(nao), m1Dtol(p1Dtol),
102 mprnt_lvl(prnt_lvl), m1Dgrid(p1Dgrid), m3Dgrid(p3Dgrid) {
103
104 if (nocc == 0) throw invalid_argument(
105 "Number of occupied orbitals cannot be 0.");
106 if (nao == 0) throw invalid_argument(
107 "Number of atomiv orbitals cannot be 0.");
108 if (!p1Dgrid.check_data_validity()) throw invalid_argument(
109 "1D Grid data not valid.");
110 if (!p3Dgrid.check_data_validity()) throw invalid_argument(
111 "3D Grid data not valid.");
112 // get number of points to be looped over, used later
113 size_t npts = m3Dgrid.get_mnpts();
114 if (!mat_fock) throw invalid_argument(
115 "Fock matrix pointer cannot be NULL.");
116 if (!mat_coeff) throw invalid_argument(
117 "Coefficient matrix pointer cannot be NULL.");
118 if (!mat_cgto) throw invalid_argument(
119 "CGTO matrix pointer cannot be NULL.");
120 if (!cube_coul) throw invalid_argument(
121 "Coulomb matrix pointer cannot be NULL.");
122
123 if (m1Dtol < 0) m1Dtol *= -1; //ensuring that tolerance is positive
124
125 //looping through the given array, copying the dat
126 mnnmo = mnocc + mnvirt;
127 size_t mnao2 = mnao * mnao;
128
129 mmat_fock = new double[mnao2];
130 for (size_t i = 0; i < mnao; i++) {
131 for (size_t j = 0; j < mnao; j++){
132 mmat_fock[i * mnao + j] = mat_fock[i * mnao + j];
133 }
134 }
135
136 mmat_coeff = new double[mnao * mnnmo];
137 for (size_t i = 0; i < nao; i++) {
138 for (size_t j = 0; j < mnnmo; j++){
139 mmat_coeff[i * mnnmo + j] = mat_coeff[i * mnnmo + j];
140 }
141 }
142
143 mmat_cgto = new double[npts * mnao];
144 for (size_t i = 0; i < npts; i++) {
145 for (size_t j = 0; j < mnao; j++){
146 mmat_cgto[i * mnao + j] = mat_cgto[i * mnao + j];
147 }
148 }
149
150 mcube_coul = new double[npts * mnao2];
151 for (size_t i = 0; i < npts; i++) {
152 for (size_t j = 0; j < mnao; j++){
153 for (size_t k = 0; k < mnao; k++){
154 mcube_coul[i * mnao2 + j * mnao + k] =
155 cube_coul[i * mnao2 + j * mnao + k];
156 }
157 }
158 }

160 check_data_validity();
