74 #pragma omp parallel for schedule(dynamic) default(none)\
75 shared(m3Dnpts, mnocc, mnvirt, mc_c, mv3Dwts)\
76 collapse(3)
77 for (size_t p = 0; p < m3Dnpts; p++){
78 for (size_t i = 0; i < mnocc; i++){

94 double compute (){
95
96
97 double e_mp2 = 0;
98#pragma omp parallel for reduction(+:e_mp2) schedule(dynamic) default(none)\
99 shared(moffset, mnpts_to_proc, m1Dnpts, m3Dnpts, mnocc, mnvirt, mv1Dwts, mmo, mm1Deps_o, mmv, mm1Deps_v, mc_c, mm1Deps_ov)\
100 collapse(2)
101 // We loop over the points on the 1D grid quadrature of
102 // the variable used to remove the energy in the denominator
103 // These are generally <10 points, so no need to mass. parralelize
104 // over this.
105 for (size_t k = 0; k < m1Dnpts; k++){
106 // Then we loop over the points on the 3D grid,
107 // which are indipendent of each other and can be parallelized
108 for (size_t p = moffset; p < moffset+mnpts_to_proc; p++){
109 // First off we scale a copy of the orbitals and integrals
110 // by the exponent of our energies, which we prescaled,
111 // as exp functions are expensive
112 double v_o_p[mnocc];
113 double v_v_p[mnvirt];
114 double m_c_p[mnocc][mnvirt];
115 for (size_t i = 0; i < mnocc; i++)
116 v_o_p[i] = mmo[p * mnocc + i]
117 * mm1Deps_o[k * mnocc + i];
118 for (size_t a = 0; a < mnvirt; a++)
119 v_v_p[a] = mmv[p * mnvirt + a]
120 * mm1Deps_v[k * mnvirt + a];
121 for (size_t i =0; i < mnocc; i++) {
122 for (size_t a = 0; a < mnvirt; a++){
123 m_c_p[i][a] =
124 mc_c [p * mnocc * mnvirt + i * mnvirt + a] *
125 mm1Deps_ov[k * mnocc * mnvirt + i * mnvirt + a];
126 }//for a
127 }//for i
128
129 // And then we loop over the inner 3D grid points.
130 // These are symmetrix, so we only need to compute
131 // one triangle of the "matrix".
132 for (size_t q = 0; q <= p; q++){
133
134 // jo, j, o, v are the names for the parts of the
135 // calculations in the original derivation
136 // by Bloomfield. But this is basically matrix
137 // multiplications, with as much cache alignment
138 // as we ca.
139 double jo = 0;
140 for (size_t a = 0; a < mnvirt; a++){
141 // we have here two intermediates we can precalc
142 // which saves some cycles
143 double tmp1 = 0;
144 double tmp2 = 0;
145 for (size_t i = 0; i < mnocc; i++){
146 // these terms are the only ones with
147 // problematic cache alignments,
148 // as we have to move in a and i
149 // direction simultaniously.
150 // Could be remedied by having a transposed
151 // copy, but this increases the data footprint.
152 tmp1 += v_o_p[i] *
153 mc_c[q * mnocc * mnvirt + i * mnvirt + a];
154 tmp2 += mmo[q*mnocc+i] * m_c_p[i][a];
155 }//for
156 jo += tmp1 * tmp2;
157 }//for a
158
159 //calc j
160 double j = 0;
161 for (size_t i = 0; i < mnocc; i++){
162 for (size_t a = 0; a < mnvirt; a++){
163 j += m_c_p[i][a]
164 * mc_c[q * mnocc * mnvirt + i * mnvirt+a];
165 }
166 }//for i
167
168 // calc o
169 double o = 0;
170 for (size_t i = 0; i < mnocc; i++){
171 o += v_o_p[i] * mmo[q * mnocc + i];
172 }//for i
173
174 // calc v
175 double v = 0;
176 for (size_t a = 0; a < mnvirt; a++){
177 v += v_v_p[a] * mmv[q * mnvirt + a];
178 }
179
180 double sum = (jo - 2 * j * o) * v;
181
182 // here we finish exploiting the symmetry and account
183 // for the double values
184 if (p != q){
185 sum *= 2.0;
186 }
187
188 // and finally we weight the energy part for this
189 // point on the 1D grid.
190 e_mp2 += sum * mv1Dwts[k];
191 }//q for
192 }//p
193 }//for k
194 return e_mp2;
195 };//calc_mp2
